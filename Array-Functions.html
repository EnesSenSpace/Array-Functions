JavaScript Array Metotları

Metot Açıklama Örnek Sonuç
Array() Bir dizi oluşturur. let numbers = Array(5, 10, 15); [5, 10, 15]
push() Bir dizinin sonuna bir veya daha fazla eleman ekler. fruits.push("Elderberry", "Fig"); ["Apple", "Banana", "Cherry", "Durian", "Elderberry", "Fig"]
pop() Bir dizinin son elemanını çıkarır. fruits.pop(); ["Apple", "Banana", "Cherry"]
shift() Bir dizinin ilk elemanını çıkarır. fruits.shift(); ["Banana", "Cherry"]
unshift() Bir dizinin başına bir veya daha fazla eleman ekler. fruits.unshift("Apple", "Apricot"); ["Apple", "Apricot", "Banana", "Cherry"]
slice() Bir dizinin belirtilen bölümünü yeni bir dizi
slice() Bir dizinin belirtilen bölümünü yeni bir dizi olarak döndürür. let numbers = [1, 2, 3, 4, 5]; let slicedNumbers = numbers.slice(1, 3); [2, 3]
splice() Bir dizinin belirtilen bölümünü değiştirir veya siler ve bu bölüme yeni elemanlar ekler. let months = ["January", "March", "April", "June"]; months.splice(1, 0, "February"); ["January", "February", "March", "April", "June"]
concat() Bir veya daha fazla dizi ile birleştirilmiş yeni bir dizi döndürür. let firstNames = ["John", "Jane"]; let lastNames = ["Doe", "Smith"]; let allNames = firstNames.concat(lastNames); ["John", "Jane", "Doe", "Smith"]
join() Bir dizinin tüm elemanlarını birleştirir ve tek bir dize olarak döndürür. let fruits = ["Apple", "Banana", "Cherry"]; let joinedFruits = fruits.join(", "); "Apple, Banana, Cherry"
reverse() Bir dizinin elemanlarını tersine çevirir. let numbers = [1, 2, 3, 4, 5]; numbers.reverse(); [5, 4, 3, 2, 1]
sort() Bir dizinin elemanlarını sıralar. let fruits = ["Banana", "Apple", "Cherry"]; fruits.sort(); ["Apple", "Banana", "Cherry"]
forEach() Bir dizinin her elemanı için belirtilen işlevi çağırır. let numbers = [1, 2, 3]; numbers.forEach(function(item, index) { console.log(item, index); }); 1 0 2 1 3 2
map() Bir dizinin her elemanı için belirtilen işlevi çağırır ve bu işlevin döndürdüğü değerleri içeren yeni bir dizi oluşturur. const numbers = [1, 2, 3, 4, 5]; const squares = numbers.map(number => number * number); console.log(squares); 1, 4, 9, 16, 25
filter() Bu metot, bir dizi içerisinde belirli bir koşulu sağlayan tüm elemanları filtrelemek için kullanılır. Filtreleme sonucunda oluşan yeni dizi, orijinal diziyi değiştirmez. const numbers = [10, 20, 30, 40, 50]; const filteredNumbers = numbers.filter(function(number) { return number > 30; }); console.log(filteredNumbers); 40, 50
reduce() Bu metot, bir dizideki elemanları tek bir değere indirgemek için kullanılır. Bu işlem sırasında her bir eleman, önceki işlemin sonucu ile birlikte yeni bir işleme tabi tutulur. const numbers = [1, 2, 3, 4, 5]; const sum = numbers.reduce(function(total, number) { return total + number; }, 0); console.log(sum); 15
find() Bu metot, bir dizide belirli bir koşulu sağlayan ilk elemanı bulmak için kullanılır. const numbers = [10, 20, 30, 40, 50]; const firstNumber = numbers.find(function(number) { return number > 30; }); console.log(firstNumber); 40
findIndex() Bu metot, bir dizide belirli bir koşulu sağlayan ilk elemanın index'ini bulmak için kullanılır. const numbers = [10, 20, 30, 40, 50]; const firstNumberIndex = numbers.findIndex(function(number) { return number > 30; }); console.log(firstNumberIndex); 3
some() Bu metot, bir dizide belirli bir koşulu sağlayan en az bir elemanın olup olmadığını kontrol etmek için kullanılır. const numbers = [10, 20, 30, 40, 50]; const hasNumberGreaterThan30 = numbers.some(function(number) { return number > 30; }); console.log(hasNumberGreaterThan30); true
every() every() metodu, bir dizi içindeki tüm öğelerin belirli bir koşulu sağlayıp sağlamadığını kontrol eder. Eğer koşul tüm öğeler için sağlanıyorsa true, aksi halde false döndürür. const numbers = [2, 4, 6, 8, 10]; const allEvenNumbers = numbers.every(num => num % 2 === 0); console.log(allEvenNumbers); true
